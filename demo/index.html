javascript

<script src="http://code.jquery.com/jquery-2.0.1.min.js"></script>
<script src="src/jquery.dataTree.min.js"></script>
<script>
    $(function () {
        $('#football').dataTree({
            delimeter: '-',
            openCSS: 'open2',
            closedCSS: 'closed2',
            endCSS: 'end2',
            opened: ['eng-epl', 'sco']
        });
    });
</script>
